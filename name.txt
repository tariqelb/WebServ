		std::cerr << "Error : initialing webserv" << std::endl;
		freedWeb(web);
		return (1);
	}
	//Set select for incomming connections
	activeSocket(web);
	while (1)
	{
		web.status = -1;
		web.tmp = web.reads;
		printf("Wait in select:\n");
		web.status = select(web.maxReadFd + 1, &web.tmp, &web.writes, &web.exceps, &tv);

			freedWeb(web);

		}
		if (web.status < 0)

			return (1);

		}
		else //(web.status == 0)
			handleConnection(web);
		handleRequest(web);

	}
	
	freedWeb(web);
	
	return (0);
}
